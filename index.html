                    int main() {
                        int a, b, c;
                        int largest;
                    
                        // Input three numbers
                        printf("Enter three numbers: ");
                        scanf("%d %d %d", &a, &b, &c);
                    
                        // Determine the largest number using nested if-else
                        if (a > b) {
                            if (a > c) {
                                largest = a;  // a is greater than both b and c
                            } else {
                                largest = c;  // c is greater than a and b
                            }
                        } else {
                            if (b > c) {
                                largest = b;  // b is greater than both a and c
                            } else {
                                largest = c;  // c is greater than a and b
                            }
                        }
                    
                        // Output the result
                        printf("The largest number is: %d\n", largest);
                    
                        return 0;
                    }
